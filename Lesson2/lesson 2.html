			var a = 1 , b = 1 , c , d;
			c = ++ a ; alert ( c ); // 2 - префиксная форма инкрементирования
			d = b ++; alert ( d ); // 1 - постфиксная форма инкрементирования. Далее в памяти присвоится значение 2.
			c = ( 2 + ++ a ); alert ( c ); // 5 - префиксная форма инкрементирования. К ранее присвоенной 2 прибавится 1 и результат а = 3 => ответ 5
			d = ( 2 + b ++); alert ( d ); // 4 - постфиксная форма инкрементирования. К ранее присвоенной 2 прибавится 1 после вывода переменной
			alert ( a ); // 3 - после вывод тех значений, которые остались в памяти.
			alert ( b ); // 3 - после вывод тех значений, которые остались в памяти.
